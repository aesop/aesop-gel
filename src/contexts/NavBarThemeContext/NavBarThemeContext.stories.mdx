# useNavBarTheme

The `NavBarThemeContextProvider` component and `useNavBarThemeContext` consumer hook are used to set and consume global values for site theme.

<Canvas>
  <Story name="NavBarThemeContextProvider">
    {() => {
      const ChildComponent = () => {
        return (
          <pre>
            <code>{JSON.stringify(useNavBarThemeContext())}</code>
          </pre>
        );
      };
      return (
        <NavBarThemeContextProvider navigationAndLogoTheme="light">
          <ChildComponent />
        </NavBarThemeContextProvider>
      );
    }}
  </Story>
</Canvas>

## Usage

```jsx
// ComponentA.js
import { NavBarThemeContextProvider } from '~/contexts';
import { useNavBarTheme } from '~/customHooks';
import ComponentB from '~/ComponentB';

const ComponentA = () => (
  <NavBarThemeContextProvider
    value={useNavBarTheme({ navigationAndLogoTheme: 'light' })}
  >
    <ComponentB />
  </NavBarThemeContextProvider>
);

// ComponentB.js
import { useNavBarThemeContext } from '~/contexts';

const { loginAndCartTheme, navigationAndLogoTheme } = useNavBarThemeContext();
const ComponentB = () => {
  return (
    <div>
      loginAndCartTheme: {loginAndCartTheme}, // Default value 'dark'
      navigationAndLogoTheme: {navigationAndLogoTheme}. // Custom value 'light'
    </div>
  );
};
```
