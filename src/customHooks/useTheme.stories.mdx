# useTheme

The `ThemeContextProvider` method, `useThemeContext` and `useTheme` hooks are used to set and consume global values for site theme.

<Preview>
  <Story name="Base hook">
    {() => {
      const ChildComponent = () => {
        return (
          <pre><code>{JSON.stringify(useThemeContext())}</code></pre>
        );
      };
      return (
        <ThemeContextProvider value={useTheme({ navigationAndLogoTheme: 'light' })}>
          <ChildComponent />
        </ThemeContextProvider>
      );
    }}
  </Story>
</Preview>

## Usage

```
// ComponentA.js
import { ThemeContextProvider } from '~/contexts';
import { useTheme } from '~/customHooks';
import ComponentB from '~/ComponentB';

const ComponentA = () => (
  <ThemeContextProvider value={useTheme({ navigationAndLogoTheme: 'light' })}>
    <ComponentB />
  </ThemeContextProvider>
)

// ComponentB.js
import { useThemeContext } from '~/contexts';

const { loginAndCartTheme, navigationAndLogoTheme } = useThemeContext();
const ComponentB = () => {
  return (
    <div>
      loginAndCartTheme: {loginAndCartTheme}, // Default value
      navigationAndLogoTheme: {navigationAndLogoTheme}. // Custom value
    </div>
  );
};
```
